Base usage: error 利用所有权机制化解了一次并发危机

use std::thread;
let mut x = vec![1, 2, 3, 4];
thread::spawn(|| {  // Error: may outlive borrowed value `x`
    x.push(1); // 子线程修改x
});
x.push(2); // 父线程修改xRun

let mut x = vec![1, 2, 3, 4];
thread::spawn(move || { // x实现了Sync和Send
    x.push(1);
});
//x.push(2); // 父线程不允许修改xRun
