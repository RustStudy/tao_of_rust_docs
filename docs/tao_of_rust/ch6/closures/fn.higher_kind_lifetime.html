}
impl<'a, T: Debug> DoSomething<T> for &'a usize {
    fn do_sth(&self, value: T) {
        println!("{:?}", value);
    }
}
fn foo<'a>(b: Box<DoSomething<&'a usize>>) {
    let s: usize = 10;
    b.do_sth(&s) // error[E0597]: `s` does not live long enough
}

impl<F> Pick<F>
    where F: Fn(&(u32, u32)) -> &u32
    {
    fn call(&self) -> &u32 {
        (self.func)(&self.data)
}
}
